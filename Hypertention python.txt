"""### Modeling Interaction Effect: Checkup Rate and No Insurance

To explicitly test if the effect of preventive care (`checkup_rate`) on hypertension differs based on insurance status (`no_insurance`), we can include an interaction term in our regression model. An interaction term captures how the effect of one predictor variable changes at different levels of another predictor variable.
"""

import statsmodels.formula.api as smf

# Create a new model with an interaction term
# The interaction between checkup_rate and no_insurance is represented by 'checkup_rate * no_insurance'

# Ensure df_pivot is available and correctly structured
# (Assuming it's available from previous cells)

# It's good practice to re-define X and y for clarity in a new modeling step
# We will use the formula API for easier interaction term specification
model_interaction = smf.ols(
    "hypertension ~ checkup_rate + no_insurance + food_insecurity + obesity + no_physical_activity + smoking + checkup_rate:no_insurance",
    data=df_pivot
).fit()

print("\n=== OLS Regression with Interaction Term (checkup_rate * no_insurance) ===")
print(model_interaction.summary())

print("\nInterpretation of the interaction term (checkup_rate:no_insurance):\n")
print("A statistically significant interaction term suggests that the effect of 'checkup_rate' on 'hypertension' changes as 'no_insurance' changes (and vice versa).")
print("If the interaction term is positive, it means the effect of one variable becomes more positive (or less negative) as the other increases.")
print("If the interaction term is negative, it means the effect of one variable becomes more negative (or less positive) as the other increases.")

import pandas as pd

import matplotlib.pyplot as plt

# =========================
# 2. LOAD DATA

# =========================
# 3. FILTER RELEVANT VARIABLES
# =========================
variables_of_interest = [
    "High blood pressure among adults",
    "Visits to doctor for routine checkup within the past year among adults",
    "Current lack of health insurance among adults aged 18-64 years",
    "Obesity among adults",
    "No leisure-time physical activity among adults",
    "Current cigarette smoking among adults"
]

df = df[df["Measure"].isin(variables_of_interest)]

# =========================
# 4. CLEAN DATA
# =========================
df = df[["LocationName", "Year", "Measure", "Data_Value"]]
df = df.dropna()

# Pivot to wide format
df_wide = df.pivot_table(
    index=["LocationName", "Year"],
    columns="Measure",
    values="Data_Value"
).reset_index()

# Rename columns for easier use
df_wide = df_wide.rename(columns={
    "High blood pressure among adults": "hypertension",
    "Visits to doctor for routine checkup within the past year among adults": "preventive_access",
    "Current lack of health insurance among adults aged 18-64 years": "uninsured",
    "Obesity among adults": "obesity",
    "No leisure-time physical activity among adults": "inactivity",
    "Current cigarette smoking among adults": "smoking"
})

# Drop missing
df_wide = df_wide.dropna()

# =========================
# 5. CREATE VARIABLES
# =========================
# Convert preventive access into "limited access"
df_wide["limited_access"] = 100 - df_wide["preventive_access"]

# Standardize variables (optional but improves modeling)
for col in ["limited_access", "uninsured", "obesity", "inactivity", "smoking"]:
    df_wide[col + "_z"] = (df_wide[col] - df_wide[col].mean()) / df_wide[col].std()

# =========================
# 6. REGRESSION MODEL WITH INTERACTION
# =========================
model = smf.ols(
    "hypertension ~ limited_access_z * uninsured_z + obesity_z + inactivity_z + smoking_z",
    data=df_wide
).fit()

print(model.summary())
